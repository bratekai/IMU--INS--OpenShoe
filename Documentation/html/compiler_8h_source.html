00609         if (is_constant(x))

00617                         (x) & (1ULL << 24) ? 24 :
00618                         (x) & (1ULL << 23) ? 23 :
00619                         (x) & (1ULL << 22) ? 22 :
00620                         (x) & (1ULL << 21) ? 21 :
00621                         (x) & (1ULL << 20) ? 20 :
00622                         (x) & (1ULL << 19) ? 19 :
00623                         (x) & (1ULL << 18) ? 18 :

00627                         (x) & (1ULL << 14) ? 14 :
00628                         (x) & (1ULL << 13) ? 13 :
00629                         (x) & (1ULL << 12) ? 12 :
00630                         (x) & (1ULL << 11) ? 11 :
00631                         (x) & (1ULL << 10) ? 10 :
00632                         (x) & (1ULL <<  9) ?  9 :
00633                         (x) & (1ULL <<  8) ?  8 :
